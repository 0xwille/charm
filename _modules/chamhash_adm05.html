[docs]    def paramgen(self, secparam=1024):
        group.paramgen(secparam)
        g, x = group.randomGen(), group.random() # g, [1,q-1]
        y = g ** x
        
        if debug:
            print("Public params")
            print("g =>", g); print("y =>", y)
        
        pk = { 'g':g, 'y':y }
        sk = { 'x':x }        
        return (pk, sk)
    


[docs]    def hash(self, pk, m, r = 0):
        p,q = group.p, group.q
        if r == 0: r = group.random()
        s = group.random()        
        e = group.hash(m, r)
        
        C = r - (((pk['y'] ** e) * (pk['g'] ** s)) % p) % q
        return C
